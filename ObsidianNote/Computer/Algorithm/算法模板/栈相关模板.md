
# 中缀转后缀表达式（逆波兰表达式）模板
```cpp
#include <iostream>
#include <stack>
#include <string>
#include <unordered_map>
using namespace std;

// 运算符优先级表
int precedence(char op) {
    if (op == '+' || op == '-') return 1;
    if (op == '*' || op == '/') return 2;
    return 0;
}

// 判断是否是运算符
bool isOperator(char c) {
    return c == '+' || c == '-' || c == '*' || c == '/';
}

string infixToPostfix(const string& infix) {
    stack<char> ops;  // 运算符栈
    string postfix;

    for (char c : infix) {
        if (isdigit(c)) {
            postfix += c;  // 数字直接加入后缀表达式
        }
        else if (c == '(') {
            ops.push(c);
        }
        else if (c == ')') {
            while (!ops.empty() && ops.top() != '(') {
                postfix += ops.top();
                ops.pop();
            }
            ops.pop(); // 弹出 '('
        }
        else if (isOperator(c)) {
            while (!ops.empty() && precedence(ops.top()) >= precedence(c)) {
                postfix += ops.top();
                ops.pop();
            }
            ops.push(c);
        }
    }

    while (!ops.empty()) {
        postfix += ops.top();
        ops.pop();
    }

    return postfix;
}

int main() {
    string infix = "3+(4*5)-6/2";
    string postfix = infixToPostfix(infix);
    cout << "后缀表达式: " << postfix << endl;
    return 0;
}
```
# 后缀表达式求值模板
```cpp
#include <iostream>
#include <stack>
#include <string>
#include <cstdlib>  // atoi
using namespace std;

int evaluatePostfix(const string& postfix) {
    stack<int> st;

    for (char c : postfix) {
        if (isdigit(c)) {
            st.push(c - '0');
        } else {
            int b = st.top(); st.pop();
            int a = st.top(); st.pop();
            switch(c) {
                case '+': st.push(a + b); break;
                case '-': st.push(a - b); break;
                case '*': st.push(a * b); break;
                case '/': st.push(a / b); break;
            }
        }
    }

    return st.top();
}

int main() {
    string postfix = "345*+62/-";
    int result = evaluatePostfix(postfix);
    cout << "表达式结果: " << result << endl;
    return 0;
}
```
# 括号匹配（有效括号）判断模板
```cpp
#include <iostream>
#include <stack>
#include <string>
using namespace std;

bool isValid(string s) {
    stack<char> st;
    for (char c : s) {
        if (c == '(' || c == '[' || c == '{') {
            st.push(c);
        } else {
            if (st.empty()) return false;
            char top = st.top();
            if ((c == ')' && top != '(') ||
                (c == ']' && top != '[') ||
                (c == '}' && top != '{')) {
                return false;
            }
            st.pop();
        }
    }
    return st.empty();
}

int main() {
    string s = "([{}])";
    cout << (isValid(s) ? "有效" : "无效") << endl;
    return 0;
}
```
